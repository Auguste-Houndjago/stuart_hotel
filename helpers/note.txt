  libelleC VARCHAR(45) NULL,
  nbplace INT NULL,
  etage VARCHAR(45) NULL,
  numE INT NULL,
  PRIMARY KEY (codeC),
  FOREIGN KEY (numE) REFERENCES eleves(numE)
) ENGINE=InnoDB;

  numP INT NOT NULL AUTO_INCREMENT,
  nomP VARCHAR(45) NULL,
  prenomP VARCHAR(45) NULL,
  datenessP DATE NULL,
  adP VARCHAR(45) NULL,
  diplome VARCHAR(45) NULL,
  ancientee INT NULL,
  matiere VARCHAR(45) NULL,

) ENGINE=InnoDB;

  numP INT NOT NULL,
  numE INT NOT NULL,
  FOREIGN KEY (numP) REFERENCES professeur(numP),
  FOREIGN KEY (numE) REFERENCES eleves(numE),
